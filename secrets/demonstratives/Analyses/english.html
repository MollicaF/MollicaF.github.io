Rubio-Fernandez (2022) Exp 1

design = expand.grid(spos=c(0,1,2,3),
            lpos=c(0,1,2,3)) %>%
  mutate(Item=1:n()) %>%
  expand_grid(referent=c(0,1,2,3),
              word=c('este','aquel')) %>%
  select(Item, spos, lpos, word, referent) 

W = design %>% pull(word) %>% unique() %>% length
R = design %>% pull(referent) %>% unique() %>% length
K = design %>% pull(Item) %>% unique() %>% length

semantics_distance = design %>%
  mutate(distance = case_when(grepl('este', word) ~ -abs(referent - spos),
                              T ~ abs(referent - spos)))

semantics_person = design %>%
  mutate(distance = case_when(grepl('este', word) ~ -abs(referent - spos),
                              T ~ abs(referent - spos) + abs(referent - lpos)))

Distance Semantics Model

stan_data_distance = list(W = W,
                          R = R,
                          K = K,
                          distance = semantics_distance %>% 
                              arrange(word, referent, Item) %>% 
                              pull(distance) %>% 
                              array(dim=c(K, R, W)),
                          N_resp = d_eng %>%
                              arrange(Item) %>%
                              spread(word, y, fill = 0) %>%
                              nrow(),
                          Ki = d_eng %>%
                              arrange(Item) %>%
                              spread(word, y, fill = 0) %>% 
                              pull(Item) %>% as.integer(),
                          ref = d_eng %>%
                              arrange(Item) %>%
                              spread(word, y, fill = 0) %>% 
                              mutate(referent = referent+1) %>%
                              pull(referent) %>% 
                              as.integer(),
                          y = d_eng %>%
                              arrange(Item) %>%
                              spread(word, y, fill = 0) %>%
                              select(aquel, este) %>%
                              as.matrix() %>%
                              unname()
)

fit_dist = stan('model.stan',
                data=stan_data_distance,
                seed=67651,
                chains=4,
                cores=4,
                iter = 2000)

Individual Subject Analysis

Let’s see for each subject, which semantics is a better fit.

eng %>% select(-Coded_THIS) %>% 
    rename(Trial=Item, referent=Object_Position, lpos=Listener_Position) %>%
    mutate(spos = 0,
           word = case_when(grepl('THIS', Raw_Response) ~ 'este',
                                T ~ 'aquel')) %>%
    select(Subject, Trial, spos, lpos, referent, word) %>%
    left_join(S1_stan_dist %>% rename(dist=S1)) %>%
    left_join(S1_stan_pers %>% rename(pers=S1)) %>%
    group_by(Subject) %>%
    summarise(like_dist = sum(log(dist)),
