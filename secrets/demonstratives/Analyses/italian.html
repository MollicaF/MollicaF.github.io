Vizualize the Data

ita = read.csv('Data/EXP1_DISTANCE_ITALIAN.csv') %>%
    filter(!is.na(Object_Position))

plt_data = ita %>%
  group_by(Object_Position, Listener_Position, Raw_Response) %>%
  summarise(N=n()) %>%
  group_by(Object_Position, Listener_Position) %>%
  mutate(p = N / sum(N)) %>%
  ggplot(aes(Listener_Position, Object_Position)) +
    facet_wrap(~Raw_Response) +
    geom_tile(aes(fill=p)) +
    scale_fill_steps2(limits=c(0, 1), n.breaks=10) +
    ylab('Referent Position') +
    xlab('Listener Position') +
    ggtitle('Empirical S1') +
    guides(fill='none') +
    theme_bw(base_size = 12)

d_ita = ita %>% 
    rename(Trial=Item, referent=Object_Position, lpos=Listener_Position) %>%
    mutate(spos = 1,

                                grepl('ELL', Raw_Response) ~ 'aquel')) %>%
    select(Subject, Trial, spos, lpos, referent, word) %>%
    group_by(spos, lpos, word, referent) %>%
    summarise(y = n()) %>%
    left_join(design) %>%
    drop_na() %>%
    group_by(Item, referent) %>%
    mutate(N=sum(y)) %>%

Distance Semantics Model

stan_data_distance = list(W = W,
                          R = R,
                          K = K,
                          distance = semantics_distance %>% 
                              arrange(word, referent, Item) %>% 
                              pull(distance) %>% 
                              array(dim=c(K, R, W)),
                          N_resp = d_ita %>%
                              arrange(Item) %>%
                              spread(word, y, fill = 0) %>%
                              nrow(),
                          Ki = d_ita %>%
                              arrange(Item) %>%
                              spread(word, y, fill = 0) %>% 
                              pull(Item) %>% as.integer(),
                          ref = d_ita %>%
                              arrange(Item) %>%
                              spread(word, y, fill = 0) %>% 
                              pull(referent) %>% 
                              as.integer(),
                          y = d_ita %>%
                              arrange(Item) %>%
                              spread(word, y, fill = 0) %>%
                              select(aquel, este) %>%
                              as.matrix() %>%
                              unname()
)

fit_dist = stan('model.stan',
                data=stan_data_distance,
                seed=67651,
                chains=4,
                cores=4,
                iter = 2000)

Individual Analysis

ita %>% 
    rename(Trial=Item, referent=Object_Position, lpos=Listener_Position) %>%
    mutate(spos = 1,
           word = case_when(grepl('EST', Raw_Response) ~ 'este',
                                grepl('ELL', Raw_Response) ~ 'aquel')) %>%
    select(Subject, Trial, spos, lpos, referent, word) %>%
    left_join(S1_stan_dist %>% rename(dist=S1)) %>%
    left_join(S1_stan_pers %>% rename(pers=S1)) %>%
    group_by(Subject) %>%
    summarise(like_dist = sum(log(dist)),
              like_pers = sum(log(pers))) %>%
    mutate(d_factor = like_dist - like_pers,
           Model = case_when(d_factor > 0 ~ 'Distance',
                             d_factor < 0 ~ 'Person',
                             T ~ NA)) %>% 
    group_by(Model) %>%
    summarise(N=n())
